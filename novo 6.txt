var BASE_FIADOR = '../fes-web/servicos/contratofies/manutencaofiador';

window.FiadorModalControle = Backbone.View.extend({
  callback: null,
  modelFiador: null,

  initialize: function () {
    var that = this;

    // garante clone (evita mexer no model original)
    var cloneFiador = JSON.stringify(this.model.toJSON ? this.model.toJSON() : this.model);
    that.modelFiador = new Fiador(JSON.parse(cloneFiador));

    $.when(
      $.getScript(BASE_FIADOR + '/controle/FiadorControle.js'),
      $.getScript(BASE_FIADOR + '/controle/FiadorConjugeControle.js'),
      $.getScript(BASE_FIADOR + '/controle/FiadorEnderecoControle.js'),
      $.getScript(BASE_FIADOR + '/controle/FiadorContatoControle.js')
    ).done(function () {

      $.get(BASE_FIADOR + '/visao/FiadorModal.html').done(function (data) {
        that.template = _.template(data);
        that.render();
      });

    }).fail(function (xhr, status, err) {
      console.error('[FiadorModalControle] FAIL carregar deps', status, err);
    });
  },

  render: function () {
    // renderiza o "esqueleto" das abas
    $(this.el).html(this.template());

    var _this = this;

    // LIMPA e instancia subviews apontando pro EL da aba
    $('#tabDadosFiador').empty();
    $('#tabConjuge').empty();
    $('#tabEndereco').empty();
    $('#tabContato').empty();

    new FiadorControle({
      el: $('#tabDadosFiador'),
      model: _this.modelFiador
    });

    new FiadorConjugeControle({
      el: $('#tabConjuge'),
      model: _this.modelFiador
    });

    new FiadorEnderecoControle({
      el: $('#tabEndereco'),
      model: _this.modelFiador
    });

    new FiadorContatoControle({
      el: $('#tabContato'),
      model: _this.modelFiador
    });

    // título/botão (se existirem no modal “pai”)
    if ($('#modalLabel').length) $('#modalLabel').html('Dados do Fiador');
    if ($('#btnSalvar').length) $('#btnSalvar').html('Salvar mudanças');

    return this;
  }
});
