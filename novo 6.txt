window.FiadorControle = Backbone.View.extend({

    events: {
        "blur input": "updateModel",
        "change select": "updateModel"
    },

    initialize: function () {
        console.log("call -> FiadorControle -> initialize");
        this.render();
    },

    render: function () {
        console.log("call -> FiadorControle -> render");
        var that = this;

        // ===== AUTOCOMPLETE PROFISSÃO (CORRIGIDO) =====
        $("#profissao").autocomplete({

            minLength: 3,
            delay: 300,

            source: function (request, response) {

                var $input = $("#profissao");

                // 🔒 trava só o campo (não usa modal)
                $input.prop("readonly", true)
                      .addClass("ui-autocomplete-loading");

                $.ajax({
                    url: "../fes-web/emprest/profissao/consultar", // mantenha sua URL real
                    type: "GET",
                    dataType: "json",
                    cache: false,
                    data: {
                        termo: request.term
                    }
                })
                .done(function (data) {

                    console.log("autocomplete retorno:", data);

                    var lista = data;

                    // caso o backend devolva objeto envelopado
                    if (data && data.lista) lista = data.lista;
                    if (data && data.retorno) lista = data.retorno;

                    var items = (lista || []).map(function (item) {
                        return {
                            label: item.descricao || item.nome || item.label || item,
                            value: item.descricao || item.nome || item.value || item,
                            objeto: item
                        };
                    });

                    response(items);
                })
                .fail(function (xhr) {
                    console.error("Erro autocomplete profissão:", xhr.responseText);
                    response([]);
                })
                .always(function () {
                    // 🔓 libera o campo
                    $input.prop("readonly", false)
                          .removeClass("ui-autocomplete-loading");
                });
            },

            select: function (event, ui) {
                if (ui && ui.item) {
                    $("#profissao").val(ui.item.value);
                }
                return false;
            }
        });

        return this;
    },

    updateModel: function (e) {
        console.log("call -> FiadorControle -> updateModel");

        var obj = {};
        obj[e.target.name] = e.target.value;
        this.model.set(obj);
    }

});
